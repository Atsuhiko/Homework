fig = plt.figure()
axes1 = fig.add_subplot(2,2,1)
axes2 = fig.add_subplot(2,2,2)
axes3 = fig.add_subplot(2,2,3)

# タイトル
#fig.suptitle("Iris parameters - Histogram") #サブタイトルと重なるので非表示にした
axes1.set_title("a1 vs a2")

axes3.set_title("a1 vs a4")

# 散布図の描画
